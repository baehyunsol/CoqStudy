Functional Programming

Map, Filter, Fold

1Fixpoint filter {X : Type} (test : X -> bool) (l : list X) : list X :=

2  match l with

3  | nil => nil

4  | h :: t => if test h

5    then h :: (filter test t)

6    else filter test t

7  end.

8

9Fixpoint map {X Y : Type} (f : X -> Y) (l : list X) : list Y :=

10  match l with

11  | [] => []

12  | h :: t => (f h) :: (map f t)

13  end.

14

15Fixpoint fold {X Y: Type} (f : X -> Y -> Y) (l : list X) (b : Y) : Y :=

16  match l with

17  | [] => b

18  | h :: t => f h (fold f t b)

19  end.

Currying

pair와 prod의 정의도 포함시켰습니다.

1Inductive prod (X Y : Type) : Type :=

2| pair (x : X) (y : Y).

3

4Arguments pair {X} {Y}.

5Notation "( x , y )" := (pair x y).

6Notation "X * Y" := (prod X Y) : type_scope.

7

8Definition fst {X Y : Type} (p : X * Y) : X :=

9  match p with

10  | (x, y) => x

11  end.

12Definition snd {X Y : Type} (p : X * Y) : Y :=

13  match p with

14  | (x, y) => y

15  end.

16

17Definition prod_curry {X Y Z : Type}

18  (f : X * Y -> Z) (x : X) (y : Y) : Z := f (x, y).

19

20Definition prod_uncurry {X Y Z : Type}

21  (f : X -> Y -> Z) (x : X * Y) : Z := f (fst x) (snd x).

2Fixpoint eqb (n m : nat) : bool :=

3  match n with

4  | O => match m with

5    | O => true

6    | _ => false

7    end

8  | S n' => match m with

9    | O => false

10    | S m' => eqb n' m'

11    end

12  end.

13

14Fixpoint leb (m n : nat) : bool :=

15  match m with

16  | O => true

17  | S m' => match n with

18    | O => false

19    | S n' => leb m' n'

20    end

21  end.

22

24Notation "x <=? y" := (leb x y) (at level 70) : nat_scope.

1Inductive list (X : Type) : Type :=

2  | nil

3  | cons (x : X) (l : list X).

4

5Arguments nil {X}.

6Arguments cons {X}.

7

8Notation "x :: l" := (cons x l)

10Notation "[ ]" := nil.

13Fixpoint append {X : Type} (l1 l2 : list X) : list X :=

14  match l1 with

15  | nil => l2

16  | h :: t => h :: (append t l2)

17  end.
